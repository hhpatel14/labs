[source,bash,role=execute]
----

Verify OADP Version
[source,bash,role=execute]
----
oc get clusterserviceversion -n openshift-adp
----

Verify OADP Velero Custom Resource configuration
[source,bash,role=execute]
----
oc get velero -n openshift-adp -oyaml
----

[source,bash,role=execute]
----
oc get deployments -n openshift-adp
